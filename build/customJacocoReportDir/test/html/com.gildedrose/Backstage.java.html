package com.gildedrose;

public class Backstage extends Item{

	protected static final int SELLIN_BACKSTAGE_CAP1 = 10;
	protected static final int SELLIN_BACKSTAGE_CAP2 = 5;

    public Backstage(Item itemParameter) {
        super(itemParameter.name, itemParameter.sellIn, itemParameter.quality);
    }

	@Override
	public void update(){
		this.updateSellIn();
		this.updateQuality();
	}

	private void updateSellIn(){
		this.sellIn -= SELLIN_DECREASE;
	}

	private void updateQuality(){
		if(this.sellIn <= SELLIN_BACKSTAGE_CAP1 && this.quality < MAX_QUALITY){
			this.quality += QUALITY_INCREASE;
		}
		if(this.sellIn <= SELLIN_BACKSTAGE_CAP2 && this.quality < MAX_QUALITY){
			this.quality += QUALITY_INCREASE;
		}
		if(this.quality < MAX_QUALITY){
			this.quality += QUALITY_INCREASE;
		}
		if(this.sellIn <= SELLIN_CAP){
			this.quality = MIN_QUALITY;
		}
	}
}
