package com.gildedrose;

class GildedRose {
	protected static final int MIN_QUALITY = 0;
	protected static final int MAX_QUALITY = 50;
	protected static final int QUALITY_DECREASE = 1;
	protected static final int QUALITY_INCREASE = 1;
	protected static final int SELLIN_CAP = 0;
	protected static final int SELLIN_DECREASE = 1;
	protected static final int SELLIN_BACKSTAGE_CAP1 = 11;
	protected static final int SELLIN_BACKSTAGE_CAP2 = 6;

	Item[] items;

	GildedRose(Item[] items) {
		this.items = items;
	}

	public void updateQuality() {
		for (int i = 0; i < items.length; i++) {
			if (!items[i].name.equals("Aged Brie") && !items[i].name.equals("Backstage passes to a TAFKAL80ETC concert")) {
				if (items[i].quality > MIN_QUALITY ) {
					if (!items[i].name.equals("Sulfuras, Hand of Ragnaros")) {
						items[i].quality = items[i].quality - QUALITY_DECREASE;
					}
				}
			} 
			else {
				if (items[i].quality < MAX_QUALITY) {
					items[i].quality = items[i].quality + QUALITY_INCREASE;

					if (items[i].name.equals("Backstage passes to a TAFKAL80ETC concert")) {
						if (items[i].sellIn < SELLIN_BACKSTAGE_CAP1) {
							items[i].quality = items[i].quality + QUALITY_INCREASE;
						}

						if (items[i].sellIn < SELLIN_BACKSTAGE_CAP2) {
							items[i].quality = items[i].quality + QUALITY_INCREASE;
						}
					}
				}
			}

			sellInUpdate(i);

			if (items[i].sellIn < SELLIN_CAP) {
				if (!items[i].name.equals("Aged Brie")) {
					if (!items[i].name.equals("Backstage passes to a TAFKAL80ETC concert")) {
						if (items[i].quality > 0) {
							if (!items[i].name.equals("Sulfuras, Hand of Ragnaros")) {
								items[i].quality = items[i].quality - QUALITY_DECREASE;
							}
						}
					} else {
						items[i].quality = items[i].quality - items[i].quality;
					}
				} else {
					if (items[i].quality < MAX_QUALITY) {
						items[i].quality = items[i].quality + QUALITY_INCREASE;
					}
				}
			}
		}
	}

	public void sellInUpdate(int index){
		if (!items[index].name.equals("Sulfuras, Hand of Ragnaros")) {
				items[index].sellIn = items[index].sellIn - SELLIN_DECREASE;
			}
	}
}
